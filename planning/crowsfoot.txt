  Tier {
    int tier_id PK
    string tier_name UK
    decimal cost
    int message_limit
    char pro_access
  }

  User {
    int user_id PK
    int tier_id FK
    string name
    string email UK
    date creation_date
    string language
  }

  BillingRecord {
    int user_id PK, FK
    string payment_method
    string billing_address
  }

  Invoice {
    int invoice_id PK
    int user_id FK
    int tier_id FK
    date invoice_date
    decimal amount
    string payment_status
  }

  Workspace {
    int workspace_id PK
    int owner_id FK
    string name
    string visibility
    datetime created_at
  }

  WorkspaceMember {
    int workspace_id PK, FK
    int user_id PK, FK
    string role
    datetime joined_at
  }

  Persona {
    int persona_id PK
    int owner_user_id FK
    string name
    string instructions
    datetime created_at
  }

  Conversation {
    int conversation_id PK
    int user_id FK
    int workspace_id FK
    int persona_id FK
    string title
    datetime start_time
    string status
    string persona_name_snapshot
    string persona_instr_snapshot
  }

  Message {
    int message_id PK
    int conversation_id FK
    string sender_role
    datetime time_sent
    string content
  }

  MessageFeedback {
    int feedback_id PK
    int message_id FK, UK
    string rating
    string comment_text
    datetime submitted_at
  }

  MessageBookmark {
    int user_id PK, FK
    int message_id PK, FK
    datetime bookmarked_time
  }

  PromptTemplate {
    int template_id PK
    int owner_user_id FK
    string name
    string text
    string category
    string visibility
    datetime created_at
  }

  WorkspacePromptTemplate {
    int workspace_id PK, FK
    int template_id PK, FK
    datetime shared_at
  }

  SupportAgent {
    int agent_id PK
    string name
    string email UK
    string specialty
    date hired_date
  }

  SupportTicket {
    int ticket_id PK
    int user_id FK
    int agent_id FK
    string topic_name
    datetime opened_time
    datetime closed_time
    string status
  }
